A library to craft composable, modular, and testable command line applications

Craft It

~/www/my-cli-app/cmd.js
var testCommand = command(
  alias('test', 't'),
  handler(function (context) {
    console.log('Testing cli.js!')
  })
)

var testFlag = flag(
  alias('--test', '-h'),
  handler(function (value, context) {
    console.log('Flag value:', value)
  })
)

run(
  process.argv.slice(2),
  app(
    testCommand(),
    testFlag()
  )
)

Execute It

~/www/cli-foo $: _
